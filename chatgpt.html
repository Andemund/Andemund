    <script src="https://cdn.jsdelivr.net/npm/@openai/api-js@1.1.0/dist/openai-api.min.js"></script>

    <script>

      const openai = new OpenAI(api_key = 'YOUR_API_KEY');

      function getChatResponse() {

        const userInput = document.getElementById("user-input").value;

        openai.complete({

          engine: 'davinci',

          prompt: userInput,

          maxTokens: 150,

          n: 1,

          temperature: 0.7,

          stop: ['\n']

        }).then(function(response) {

          const chatOutput = document.getElementById("chat-output");

          const chatResponse = response.data.choices[0].text;

          chatOutput.innerHTML += "<p><strong>您：</strong> " + userInput + "</p>";

          chatOutput.innerHTML += "<p><strong>ChatGPT：</strong> " + chatResponse + "</p>";

        }).catch(function(error) {

          console.error(error);

        });

      }

    </script>
